3    import android.content.Context;

5    import android.os.Bundle;

10   import android.widget.TextView; 
11    
12   import com.google.android.gms.ads.AdLoader; 
13   import com.google.android.gms.ads.doubleclick.PublisherAdRequest; 
14   import com.google.android.gms.ads.formats.NativeCustomTemplateAd; 
15    
16    
17   public class MainActivity extends AppCompatActivity { 
18    
19       @Override 
20       protected void onCreate(Bundle savedInstanceState) { 
21           super.onCreate(savedInstanceState); 
22           setContentView(R.layout.activity_main); 
23    
24           AdLoader adLoader = new AdLoader.Builder(getApplicationContext(), 
25                   getString(R.string.native_ad_unit_id)) 
26                   .forCustomTemplateAd(getString(R.string.native_template_id), 
27                           new NativeCustomTemplateAd.OnCustomTemplateAdLoadedListener() { 
28                               @Override 
29                               public void onCustomTemplateAdLoaded(NativeCustomTemplateAd ad) { 
30                                   ViewGroup adView = (ViewGroup) findViewById(R.id.adView); 
31                                   displayCustomTemplateAd(adView, ad); 
32                               } 
33                           }, 
34                           null) 
35                   .build(); 
36    
37           adLoader.loadAd(new PublisherAdRequest.Builder().build()); 
38       } 
39    
40       private void displayCustomTemplateAd (ViewGroup parent, final NativeCustomTemplateAd ad) { 
41           // Inflate a layout and add it to the parent ViewGroup. 
42           LayoutInflater inflater = (LayoutInflater) parent.getContext() 
43                   .getSystemService(Context.LAYOUT_INFLATER_SERVICE); 
44           View adView = inflater.inflate(R.layout.custom_template_ad, parent); 
45    
46           // Show the custom template 
47           TextView headline = (TextView) findViewById(R.id.headline); 
48           TextView caption = (TextView) findViewById(R.id.caption); 
49           ImageView mainImage = (ImageView) findViewById(R.id.mainImage); 
50           headline.setText(ad.getText("Headline")); 
51           caption.setText(ad.getText("Caption")); 
52           mainImage.setImageDrawable(ad.getImage("MainImage").getDrawable()); 
53    
54           // Record an impression 
55           ad.recordImpression(); 
56    
57           // Handle clicks on image 
58           mainImage.setOnClickListener(new View.OnClickListener() { 
59               @Override 
60               public void onClick(View v) { 
61                   ad.performClick("MainImage"); 
62               } 
63           }); 
64       } 
65    
66       @Override 
67       public void onStart() { 
68           super.onStart(); 
69       } 
70    
71       @Override 
72       public void onStop() { 
73           super.onStop(); 
74       } 
75   }
